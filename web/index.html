#--The field `key` is optional

#--script
#!/bin/bash

WABIO_KEY=[uuid] #generate one with uuidgen|sed s/-//g
function clipboard_insert() {
xclip -selection clipboard
}

function wabio() {
curl -s -F url=$("/bin/xclip" -o) -F key=WABIO_KEY https://wab.io | tr -d '\n' | clipboard_insert 
}
